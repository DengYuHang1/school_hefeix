   	s=s*i;
   	sum+=s;
   }
   printf("1!+2!+...+10! = %ld\n",sum);
    return 0;
}

华氏和摄氏温度的转换公式为:
          C = 5/9 x (F-32)
其中，C表示摄氏温度，F表示华氏温度。
要求：华氏0℉～300℉，每隔20℉输出一个华氏温度对应的摄氏温度值。
**输入提示信息要求:无
**输入数据格式要求:无
**输出数据格式要求: "%4.0f%10.1f\n"  (第一个浮点数为华氏温度，第二个浮点数为摄氏温度)
#include <stdio.h>
int main()
{
    int    upper = 300, step = 20;
    float  fahr = 0, celsius;
    while (fahr <= upper)
    {
        celsius =  (fahr-32)*5/9;
        printf("%4.0f%10.1f\n", fahr, celsius);
        fahr = fahr + step ;
    }
    return 0;
}
